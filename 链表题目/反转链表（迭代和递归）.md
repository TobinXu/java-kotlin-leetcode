/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */

  迭代法
var reverseList = function(head) {
  let pre = null;
  let curr = null;
  while(curr !== null){
  let nextTemp = curr.next;
  curr.next = prev;
  prev = curr;
  curr = nextTemp;
  }
  return prev;
};
reverseList([1,2,3,4,5]);
